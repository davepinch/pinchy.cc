  ingredients:
    - item: "broccoli"
      unit: "head"
      amount: 1
    - item: "extra virgin olive oil"
      unit: "tablespoon"
      amount: 3
    - item: "lemon juice"

    - item: "garlic"

    - item: "Parmesan cheese"

    - "Mince the garlic"
    - "Put broccoli and minced garlic into a large bowl"

    - "Prepare baking sheet with parchment paper"
